# 记忆系统

记忆系统决定了 Agent 能不能持续保持上下文、复用历史经验、以及在多轮任务中维持稳定性。

很多 Agent 看起来“不聪明”，问题并不在模型，而在它没有记住该记住的信息，或者在错误的时机取回了错误的记忆。

## 🧠 记忆的类型

### 短期记忆（Short-term Memory）

**特点**：
- 容量有限（受限于 LLM 上下文窗口）
- 保存当前对话和最近操作
- 快速访问

**实现方式**：
```python
from langchain_community.memory import ConversationBufferMemory

# 简单的对话记忆
memory = ConversationBufferMemory(
    memory_key="chat_history",
    return_messages=True,
    max_token_limit=2000  # 限制 token 数量
)

# 保存对话
memory.save_context(
    {"input": "你好，我想了解 AI Agent"},
    {"output": "AI Agent 是..."}
)

# 获取历史
history = memory.load_memory_variables({})
```

### 长期记忆（Long-term Memory）

**特点**：
- 理论上无限容量
- 需要外部存储（向量数据库）
- 通过检索访问

**实现方式**：
```python
from langchain_community.vectorstores import Chroma
from langchain_openai import OpenAIEmbeddings

# 初始化向量存储
embeddings = OpenAIEmbeddings()
vectorstore = Chroma(
    embedding_function=embeddings,
    persist_directory="./long_term_memory"
)

# 添加记忆
vectorstore.add_texts(
    texts=["用户偏好 Python 编程", "项目截止日期是下周五"],
    metadatas=[{"type": "preference"}, {"type": "deadline"}]
)

# 检索记忆
results = vectorstore.similarity_search(
    query="用户的技术偏好",
    k=3,
    filter={"type": "preference"}
)
```

## 一个更实用的理解

记忆系统真正要解决的不是“存多少”，而是三件事：

1. 存什么
2. 什么时候取
3. 取出来后如何影响当前任务

## 🔧 记忆操作

### 1. 写入记忆

```python
class MemoryManager:
    def __init__(self, vectorstore, short_term_limit=10):
        self.vectorstore = vectorstore
        self.short_term = []
        self.limit = short_term_limit
    
    def add(self, content: str, metadata: dict = None):
        """添加记忆"""
        # 添加到短期记忆
        self.short_term.append({
            "content": content,
            "metadata": metadata or {},
            "timestamp": time.time()
        })
        
        # 维护短期记忆大小
        if len(self.short_term) > self.limit:
            # 将最旧的记忆转移到长期记忆
            old_memory = self.short_term.pop(0)
            self.consolidate(old_memory)
    
    def consolidate(self, memory):
        """将记忆转移到长期存储"""
        self.vectorstore.add_texts(
            texts=[memory["content"]],
            metadatas=[memory["metadata"]]
        )
```

### 2. 检索记忆

```python
def retrieve_memories(query: str, k: int = 5):
    """检索相关记忆"""
    # 从短期记忆搜索
    short_term_results = []
    for mem in short_term:
        similarity = calculate_similarity(query, mem["content"])
        short_term_results.append((mem, similarity))
    
    # 从长期记忆搜索
    long_term_results = vectorstore.similarity_search(query, k=k)
    
    # 合并结果
    all_results = short_term_results + long_term_results
    all_results.sort(key=lambda x: x[1], reverse=True)
    
    return all_results[:k]
```

### 3. 记忆更新

```python
def update_memory(memory_id: str, new_content: str):
    """更新记忆"""
    # 删除旧记忆
    vectorstore.delete(ids=[memory_id])
    
    # 添加新记忆
    vectorstore.add_texts(
        texts=[new_content],
        ids=[memory_id]
    )
```

### 4. 记忆遗忘

```python
def forget_old_memories(days: int = 30):
    """遗忘旧记忆"""
    cutoff_time = time.time() - (days * 24 * 3600)
    
    old_memories = vectorstore.get(
        where={"timestamp": {"$lt": cutoff_time}}
    )
    
    if old_memories:
        vectorstore.delete(ids=old_memories["ids"])
```

def compress_memory(memories: List[str]) -> str:
    """压缩多条记忆为摘要"""
    prompt = f"""
    将以下记忆压缩为简洁的摘要：
    
    {chr(10).join(memories)}
    
    摘要：
    """
    return llm.generate(prompt)

### 2. 记忆分类

```python
class MemoryCategory(Enum):
    FACT = "fact"          # 事实信息
    PREFERENCE = "preference"  # 用户偏好
    EXPERIENCE = "experience"  # 经验教训
    SKILL = "skill"        # 技能知识
    CONTEXT = "context"    # 上下文信息

def categorize_memory(content: str) -> MemoryCategory:
    """自动分类记忆"""
    prompt = f"""
    将以下记忆分类：
    
    内容：{content}
    
    类别（FACT/PREFERENCE/EXPERIENCE/SKILL/CONTEXT）：
    """
    category = llm.generate(prompt)
    return MemoryCategory(category.strip())
```

### 3. 记忆关联

```python
def link_memories(memory1: str, memory2: str, relation: str):
    """建立记忆之间的关联"""
    # 在元数据中记录关联
    metadata = {
        "linked_to": memory2,
        "relation": relation
    }
    vectorstore.update_metadata(memory1, metadata)
```

## 🎯 实践：对话记忆 Agent

```python
from langchain_openai import ChatOpenAI
from langchain_community.memory import ConversationBufferWindowMemory
from langchain.agents import AgentExecutor, create_tool_calling_agent
from langchain_core.prompts import ChatPromptTemplate, MessagesPlaceholder

# 创建带记忆的 Agent
llm = ChatOpenAI(model="gpt-4", temperature=0)
memory = ConversationBufferWindowMemory(
    k=5,  # 保留最近 5 轮对话
    return_messages=True,
    memory_key="chat_history"
)

# 创建提示词模板
prompt = ChatPromptTemplate.from_messages([
    ("system", "你是一个有帮助的助手，记住用户的偏好和历史对话"),
    MessagesPlaceholder(variable_name="chat_history"),
    ("human", "{input}"),
    MessagesPlaceholder(variable_name="agent_scratchpad"),
])

# 创建 Agent
agent = create_tool_calling_agent(llm, [], prompt)
agent_executor = AgentExecutor(
    agent=agent,
    tools=[],
    memory=memory,
    verbose=True
)

# 对话会保持上下文
agent_executor.invoke({"input": "我喜欢吃川菜"})
agent_executor.invoke({"input": "推荐一家餐厅"})
# Agent 会记住用户喜欢川菜
```

## ⚠️ 注意事项

### 1. 隐私保护

```python
def sanitize_memory(content: str) -> str:
    """清理敏感信息"""
    # 移除个人信息
    content = re.sub(r'\b\d{18}\b', '[ID]', content)  # 身份证号
    content = re.sub(r'\b1[3-9]\d{9}\b', '[PHONE]', content)  # 手机号
    
    return content
```

def check_consistency(new_memory: str, existing_memories: List[str]) -> bool:

## 🚀 进阶：分层记忆

```python
class HierarchicalMemory:
    def __init__(self):
        self.episodic = VectorStore()  # 事件记忆
        self.semantic = VectorStore()  # 语义记忆
        self.procedural = VectorStore()  # 程序记忆
    
    def store(self, content: str, memory_type: str):
        """存储记忆到对应层级"""
        if memory_type == "episodic":
            self.episodic.add_texts([content])
        elif memory_type == "semantic":
            self.semantic.add_texts([content])
        elif memory_type == "procedural":
            self.procedural.add_texts([content])
    
    def retrieve(self, query: str, memory_types: List[str] = None):
        """从指定层级检索记忆"""
        results = []
        
        for mem_type in (memory_types or ["episodic", "semantic", "procedural"]):
            store = getattr(self, mem_type)
            results.extend(store.similarity_search(query, k=3))
        
        return results
```
